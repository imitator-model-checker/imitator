\begin{frame}
 \frametitle{An Example of a Jobshop Problem}
\begin{itemize}
 \item A Jobshop problem
 \item Given a set of jobs $J_1,\dots,J_m$
 \item Given a deadline $d$
 \item Question: Can all the jobs be completed in $d$ time units?
 \pause
 \item Timed model checking gives the answer: \couleur{yes/no}
\end{itemize}

\end{frame}
